/**
 * Data-driven suppress detection (D-02, ARCH-08).
 * Rules come from OptionItem.suppresses — zero code logic.
 * When option A suppresses option B and both are selected,
 * B is excluded from visible options and an amber warning is generated.
 */
function applySuppresses(
  selectedOptions: OptionItem[]
): { visible: OptionItem[]; warnings: LocalizedText[] } {
  // Pass 1: collect all suppression target IDs
  const suppressedIds = new Set<string>();
  for (const opt of selectedOptions) {
    if (opt.suppresses) {
      for (const targetId of opt.suppresses) {
        suppressedIds.add(targetId);
      }
    }
  }

  // No suppress relationships — fast path
  if (suppressedIds.size === 0) {
    return { visible: selectedOptions, warnings: [] };
  }

  // Pass 2: filter + generate warnings
  const warnings: LocalizedText[] = [];
  const visible: OptionItem[] = [];

  for (const opt of selectedOptions) {
    if (suppressedIds.has(opt.id)) {
      // Find which option(s) suppressed this one
      const suppressors = selectedOptions.filter(
        (s) => s.suppresses?.includes(opt.id)
      );
      for (const suppressor of suppressors) {
        warnings.push({
          zh: `"${opt.label.zh}" 被 "${suppressor.label.zh}" 覆盖`,
          en: `"${opt.label.en}" overridden by "${suppressor.label.en}"`
        });
      }
      // B is excluded from visible per D-02
    } else {
      visible.push(opt);
    }
  }

  return { visible, warnings };
}
```

/**
 * CI-time validation: checks that all suppresses references point to registered options.
 * Returns string[] of errors — empty array = all valid.
 * Per D-02: suppress rules are data. This catches typos in catalog data.
 */
export function validateOptionSuppressRefs(): string[] {
  const errors: string[] = [];
  const allOptionIds = new Set<string>();
  
  // Build the set of all valid option IDs
  for (const set of getAllOptionSets()) {
    for (const option of set.options) {
      allOptionIds.add(option.id);
    }
  }
  
  // Check every option's suppresses refs
  for (const set of getAllOptionSets()) {
    for (const option of set.options) {
      if (!option.suppresses) continue;
      for (const ref of option.suppresses) {
        if (!allOptionIds.has(ref)) {
          errors.push(
            `Option "${option.id}" in set "${set.id}" suppresses unknown option ID: "${ref}"`
          );
        }
        if (ref === option.id) {
          errors.push(
            `Option "${option.id}" in set "${set.id}" suppresses itself`
          );
        }
      }
    }
  }
  
  return errors;
}
```
