```typescript
import { getBriefText, getCameraText } from "./brief";

/**
 * Generic template-driven prompt assembly (D-03).
 * Iterates brief items, looks up templateMap slot by questionId,
 * replaces "{选项}" placeholder with option text.
 *
 * Dimensions without a templateMap slot are silently skipped.
 * Dimensions with no selected text are silently skipped.
 */
export function assemblePrompt(
  brief: PromptBrief,
  templateMap: Record<string, LocalizedText>,
  locale: "zh" | "en"
): string {
  const joiner = locale === "zh" ? "；" : "; ";
  const lines: string[] = [];

  for (const item of brief.items) {
    const template = templateMap[item.questionId];
    if (!template) continue;

    const optionText = item.freeText
      ?? item.selectedOptions.map((o) => o.promptFragment[locale]).join(joiner);
    if (!optionText) continue;

    lines.push(template[locale].replace("{选项}", optionText));
  }

  return lines.join("\n");
}
```
