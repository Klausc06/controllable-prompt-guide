describe("negative prompt injection (QUAL-03)", () => {
  const imageWorkType = resolveWorkType("image_prompt");
  
  const imageSelections: PromptSelections = {
    use_case: "image_use_case:product_photo",
    subject: "image_subject:hero_product",
    scene: "image_scene:studio_env",
    composition: "image_composition:centered",
    lighting: "image_lighting:studio_clean",
    art_style: "image_art_style:product_photography",
    constraints: ["image_constraints:no_bad_anatomy", "image_constraints:no_low_quality"],
    color_palette: "image_color_palette:neutral",
    mood: "image_mood:clean_professional",
    aspect_ratio: "image_aspect_ratio:square_1_1"
  };

  it("injects negative prompt (medium tier by default) into generic image output", () => {
    const result = renderPrompt({
      workType: imageWorkType,
      targetToolId: "generic_image",
      rawIntent: "",
      selections: imageSelections
    });
    // Medium tier default per D-08
    expect(result.zhPrompt).toContain("避免：不良解剖结构");
    expect(result.zhPrompt).toContain("水印");
    expect(result.zhPrompt).toContain("扭曲面部");
    expect(result.enPrompt).toContain("avoid: bad anatomy");
    expect(result.enPrompt).toContain("watermarks");
    expect(result.enPrompt).toContain("distorted faces");
  });

  it("negative prompt appears at end of output (after main content)", () => {
    const result = renderPrompt({
      workType: imageWorkType,
      targetToolId: "generic_image",
      rawIntent: "",
      selections: imageSelections
    });
    // Negative prompt should be the last thing in the output
    expect(result.zhPrompt).toMatch(/—— 避免：.+$/);
    expect(result.enPrompt).toMatch(/-- avoid:.+$/);
  });

  it("does NOT inject negative prompt into seedance output", () => {
    const result = renderPrompt({
      workType: resolveWorkType("video_prompt"),
      targetToolId: "seedance",
      rawIntent: "",
      selections: { ... }
    });
    // Seedance output should not contain image negative prompt text
    expect(result.zhPrompt).not.toContain("避免：不良解剖结构");
  });

  it("negative prompt is present even with minimal selections", () => {
