```typescript
export function validateSafetyDefaults(targets_?: TargetToolConfig[]): string[] {
  const targets = targets_ ?? getAllTargets();
  const errors: string[] = [];

  const allOptionIds = new Set<string>();
  for (const set of getAllOptionSets()) {
    for (const option of set.options) {
      allOptionIds.add(option.id);
    }
  }

  for (const target of targets) {
    for (const optionId of target.safetyDefaults) {
      if (!allOptionIds.has(optionId)) {
        errors.push(
          `${target.id}: safetyDefault "${optionId}" does not exist in any option catalog`
        );
      }
    }
  }

  return errors;
}
```

The function accepts an optional `targets_` parameter (for injecting test targets without polluting the registry) and defaults to `getAllTargets()` from registry when not provided. It builds a Set of all option IDs across all registered option sets, then checks each target's safetyDefaults against that Set.
